#include <bits/stdc++.h>using namespace std;const int MARK = -100000001;void setRowZeroes(vector<vector<int>> &mat, int i, int m){    for(int col = 0; col < m; col++){        if(mat[i][col] != MARK){            mat[i][col] = 0;        }    }}void setColZeroes(vector<vector<int>> &mat, int j, int n){    for(int row = 0; row < n; row++){        if(mat[row][j] != MARK){            mat[row][j] = 0;        }    }}void setZeroes(vector<vector<int>> &mat) {    int n = mat.size();    int m = mat[0].size();    for(int i = 0; i < n; i++){        for(int j = 0; j < m; j++){            if(mat[i][j] == 0){                mat[i][j] = MARK;            }        }    }      int lastRow = -1, lastCol = -1;    for(int i = 0; i < n; i++){        for(int j = 0; j < m; j++){            if(mat[i][j] == MARK){                if(i != lastRow){                    setRowZeroes(mat, i, m);                    lastRow = i;                }                if(j != lastCol){                    setColZeroes(mat, j, n);                    lastCol = j;                }            }        }    }        for(int i = 0; i < n; i++){        for(int j = 0; j < m; j++){            if(mat[i][j] == MARK){                mat[i][j] = 0;            }        }    }}
